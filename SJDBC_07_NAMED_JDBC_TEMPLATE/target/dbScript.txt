


CREATE TABLE "SYSTEM"."BOOK_STORE" 
   (	"BOOKID" NUMBER(4,0), 
	"BOOKNAME" VARCHAR2(15 BYTE), 
	"AUTHOR" VARCHAR2(10 BYTE), 
	"PRICE" FLOAT(126), 
	"PUBLISHER" VARCHAR2(15 BYTE), 
	"STATUS" VARCHAR2(10 BYTE), 
	"CATEGORY" VARCHAR2(10 BYTE), 
	 PRIMARY KEY ("BOOKID"));
	 
	 CREATE SEQUENCE  "SYSTEM"."BKID_SEQ"  MINVALUE 1000 MAXVALUE 2000 INCREMENT BY 1 START WITH 1000 CACHE 20 NOORDER  NOCYCLE ;
